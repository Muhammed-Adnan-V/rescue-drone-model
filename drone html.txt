    function compose(t) {
      let m = mIdentity();
      m = mMul(m, mTranslate(t.x || 0, t.y || 0, t.z || 0));
      m = mMul(m, mRotZ((t.rz || 0) * DEG));
      m = mMul(m, mRotY((t.ry || 0) * DEG));
      m = mMul(m, mRotX((t.rx || 0) * DEG));
      const s = t.scale ?? 1;
      m = mMul(m, mScale((t.sx ?? 1) * s, (t.sy ?? 1) * s, (t.sz ?? 1) * s));
      return m;
    }

    function clamp(v, a, b) { return Math.max(a, Math.min(b, v)); }

    function hexToRgb(hex) {
      const clean = hex.replace("#", "");
      return {
        r: parseInt(clean.slice(0, 2), 16),
        g: parseInt(clean.slice(2, 4), 16),
        b: parseInt(clean.slice(4, 6), 16)
      };
    }

    function shade(hex, amount, alpha = 1) {
      const c = hexToRgb(hex);
      const r = clamp(Math.round(c.r * amount), 0, 255);
      const g = clamp(Math.round(c.g * amount), 0, 255);
      const b = clamp(Math.round(c.b * amount), 0, 255);
      return `rgba(${r},${g},${b},${alpha})`;
    }

    function box(w, h, d) {
      const x = w / 2, y = h / 2, z = d / 2;
      const v = [
        vec(-x,-y,-z), vec(x,-y,-z), vec(x,y,-z), vec(-x,y,-z),
        vec(-x,-y,z), vec(x,-y,z), vec(x,y,z), vec(-x,y,z)
      ];
      const f = [[0,1,2,3], [4,7,6,5], [0,4,5,1], [1,5,6,2], [2,6,7,3], [3,7,4,0]];
      return { vertices: v, faces: f };
    }

    function cylinder(r = 1, h = 1, segments = 24) {
      const v = [vec(0, h / 2, 0), vec(0, -h / 2, 0)];
      for (let i = 0; i < segments; i++) {
        const a = (i / segments) * TAU;
        const x = Math.cos(a) * r;
        const z = Math.sin(a) * r;
        v.push(vec(x, h / 2, z), vec(x, -h / 2, z));
      }
      const faces = [];
      for (let i = 0; i < segments; i++) {
        const n = (i + 1) % segments;
        faces.push([2 + i * 2, 2 + n * 2, 3 + n * 2, 3 + i * 2]);
        faces.push([0, 2 + i * 2, 2 + n * 2]);
        faces.push([1, 3 + n * 2, 3 + i * 2]);
      }
      return { vertices: v, faces };
    }

    function sphere(r = 1, lat = 8, lon = 16, squashY = 1) {
      const vertices = [];
      for (let y = 0; y <= lat; y++) {
        const v = y / lat;
        const phi = v * Math.PI;
        for (let x = 0; x <= lon; x++) {
          const u = x / lon;
          const theta = u * TAU;
          vertices.push(vec(Math.sin(phi) * Math.cos(theta) * r, Math.cos(phi) * r * squashY, Math.sin(phi) * Math.sin(theta) * r));
        }
      }
      const faces = [];
      for (let y = 0; y < lat; y++) {
        for (let x = 0; x < lon; x++) {
          const a = y * (lon + 1) + x;
          faces.push([a, a + 1, a + lon + 2, a + lon + 1]);
        }
      }
      return { vertices, faces };
    }

    function blade(length = 1.7, width = .32, thick = .035) {
      const l = length / 2, w = width / 2, t = thick / 2;
      const v = [
        vec(-l, -t, -w * .35), vec(l, -t, -w), vec(l, -t, w), vec(-l, -t, w * .35),
        vec(-l, t, -w * .35), vec(l, t, -w), vec(l, t, w), vec(-l, t, w * .35)
      ];
      return { vertices: v, faces: [[0,1,2,3], [4,7,6,5], [0,4,5,1], [1,5,6,2], [2,6,7,3], [3,7,4,0]] };
    }

    function chip(sizeX = .55, sizeZ = .45, h = .07) {
      return box(sizeX, h, sizeZ);
    }

    function mesh(geometry, color, local = {}, options = {}) {
      return {
        geometry,

      };

    function barBetween(x1, y1, z1, x2, y2, z2, thickness, color) {
      const dx = x2 - x1, dy = y2 - y1, dz = z2 - z1;
      const length = Math.hypot(dx, dy, dz);
      const cx = (x1 + x2) / 2, cy = (y1 + y2) / 2, cz = (z1 + z2) / 2;
      const ry = Math.atan2(dx, dz) / DEG;
      const rx = -Math.atan2(dy, Math.hypot(dx, dz)) / DEG;
      return mesh(cylinder(thickness, length, 14), color, { x: cx, y: cy, z: cz, rx: 90 + rx, ry });
    }

    function cameraBasis() {
      const cp = Math.cos(state.pitch), sp = Math.sin(state.pitch);
      const pos = vec(
        Math.sin(state.yaw) * cp * state.distance,
        sp * state.distance,
        Math.cos(state.yaw) * cp * state.distance
      );
      const target = vec(0, 0, 0);
      const forward = norm(sub(target, pos));
      const right = norm(cross(forward, vec(0, 1, 0)));
      const up = cross(right, forward);
      return { pos, target, forward, right, up };
    }

    function project(world, basis) {
      const d = sub(world, basis.pos);
      const z = dot(d, basis.forward);
      if (z < .2) return null;
      const x = dot(d, basis.right);
      const y = dot(d, basis.up);
      const f = Math.min(canvas.width, canvas.height) * .92;
      return {
        x: canvas.width / 2 + x * f / z,
        y: canvas.height / 2 - y * f / z,
        z
      };
    }

    function nodeMatrix(node) {
      const t = { ...node.transform };
      if (state.exploded) {
        t.x += node.explode.x;
        t.y += node.explode.y;
        t.z += node.explode.z;
      }
      return compose(t);
    }

    function collectFaces(basis) {
      const light = norm(vec(-.45, .85, -.36));
      const faces = [];
      nodes.forEach(node => {
        if (!node.visible) return;
        const nm = nodeMatrix(node);
        node.meshes.forEach(localMesh => {
          let lm = localMesh.local;
          if (node.id === "motors" && state.propellers) {
            const spin = state.time * 700;
            const isBlade = localMesh.geometry.vertices.length === 8 && Math.abs(localMesh.geometry.vertices[1].x - localMesh.geometry.vertices[0].x) > 1.4;
            if (isBlade) lm = mMul(localMesh.local, mRotY(spin * DEG));
          }
          const matrix = mMul(nm, lm);
          const worldVerts = localMesh.geometry.vertices.map(v => transformPoint(matrix, v));
          const cameraVerts = worldVerts.map(v => project(v, basis));
          localMesh.geometry.faces.forEach(face => {
            const ptsWorld = face.map(i => worldVerts[i]);
            const pts = face.map(i => cameraVerts[i]);
            if (pts.some(p => !p)) return;
            const normal = norm(cross(sub(ptsWorld[1], ptsWorld[0]), sub(ptsWorld[2], ptsWorld[0])));
            const toCam = norm(sub(basis.pos, ptsWorld[0]));
            if (dot(normal, toCam) < -.08 && localMesh.alpha >= .95) return;
            const brightness = .34 + Math.max(0, dot(normal, light)) * .66;
            const selectedBoost = node.id === state.selected ? .18 : 0;
            const avgZ = pts.reduce((sum, p) => sum + p.z, 0) / pts.length;
            const alpha = localMesh.alpha;
            faces.push({
              node,
              pts,
              avgZ,
              fill: shade(node.color === localMesh.color ? node.color : localMesh.color, brightness + selectedBoost, alpha),
              stroke: node.id === state.selected ? "rgba(255, 190, 83, .95)" : shade("#000000", 1, alpha > .95 ? .18 : .1),
              lineWidth: node.id === state.selected ? 1.4 : .45,
              glow: localMesh.glow
            });
          });
        });
      });
      faces.sort((a, b) => b.avgZ - a.avgZ);
      return faces;
    }

    function drawGround(basis) {
      if (!gridToggle.checked) return;
      ctx.save();
      ctx.lineWidth = 1;
      for (let i = -8; i <= 8; i++) {
        const a = project(vec(i, -1.35, -8), basis);
        const b = project(vec(i, -1.35, 8), basis);
        const c = project(vec(-8, -1.35, i), basis);
        const d = project(vec(8, -1.35, i), basis);
        ctx.strokeStyle = i === 0 ? "rgba(57,198,172,.32)" : "rgba(255,255,255,.07)";
        if (a && b) drawLine(a, b);
        if (c && d) drawLine(c, d);
      }
      ctx.restore();
    }

    function drawLine(a, b) {
      ctx.beginPath();
      ctx.moveTo(a.x, a.y);
      ctx.lineTo(b.x, b.y);
      ctx.stroke();
    }

    function drawFaces(faces) {
      faces.forEach(face => {
        ctx.beginPath();
        ctx.moveTo(face.pts[0].x, face.pts[0].y);
        for (let i = 1; i < face.pts.length; i++) ctx.lineTo(face.pts[i].x, face.pts[i].y);
        ctx.closePath();
        ctx.fillStyle = face.fill;
        ctx.fill();
        ctx.strokeStyle = face.stroke;
        ctx.lineWidth = face.lineWidth;
        ctx.stroke();
      });
    }

    function drawScene() {
      const dpr = window.devicePixelRatio || 1;
      const w = Math.max(1, canvas.clientWidth);
      const h = Math.max(1, canvas.clientHeight);
      if (canvas.width !== Math.floor(w * dpr) || canvas.height !== Math.floor(h * dpr)) {
        canvas.width = Math.floor(w * dpr);
        canvas.height = Math.floor(h * dpr);
      }

      ctx.setTransform(1, 0, 0, 1, 0, 0);
      ctx.fillStyle = "#101211";
      ctx.fillRect(0, 0, canvas.width, canvas.height);
      const grd = ctx.createLinearGradient(0, 0, 0, canvas.height);
      grd.addColorStop(0, "#171a18");
      grd.addColorStop(.58, "#101211");
      grd.addColorStop(1, "#0b0c0b");
      ctx.fillStyle = grd;
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      const basis = cameraBasis();
      drawGround(basis);
      const faces = collectFaces(basis);
      drawFaces(faces);
      updateLabels(basis);
    }

    function updateLabels(basis) {
      const showLabels = labelToggle.checked;
      nodes.forEach(node => {
        const el = node.labelEl;
        if (!el) return;
        const p = project(transformPoint(nodeMatrix(node), node.anchor), basis);
        const isSelected = node.id === state.selected;
        el.classList.toggle("selected", isSelected);
        el.style.display = showLabels && node.visible && p && (isSelected || state.exploded) ? "block" : "none";
        if (p) {
          el.style.left = `${p.x / (window.devicePixelRatio || 1)}px`;
          el.style.top = `${p.y / (window.devicePixelRatio || 1)}px`;
          el.style.zIndex = `${Math.max(1, Math.round(5000 - p.z * 10))}`;
        }
      });
    }

    function renderLoop(now) {
      state.time = now / 1000;
      if (autoRotate.checked && !state.drag) state.yaw += .004;
      drawScene();
      requestAnimationFrame(renderLoop);
    }

    function selectNode(id) {
      state.selected = id;
      renderList();
      renderEditor();
    }

    function renderList() {
      const query = state.search.trim().toLowerCase();
      elementList.innerHTML = "";
      elementDefs
        .filter(item => state.filter === "All" || item.group === state.filter || (state.filter === software && item.group === ai))
        .filter(item => !query || `${item.name} ${item.group} ${item.purpose}`.toLowerCase().includes(query))
        .forEach(item => {
          const node = nodes.get(item.id);
          const button = document.createElement("button");
          button.type = "button";
          button.className = `element-item${item.id === state.selected ? " active" : ""}`;
          button.innerHTML = `<span class="swatch" style="background:${node.color}"></span><span><span class="element-name">${item.name}</span><span class="element-kind">${item.group}</span></span>`;
          button.addEventListener("click", () => selectNode(item.id));
          elementList.appendChild(button);
        });
    }

    const controlDefs = [
      ["x", "X", -3, 3, .01],
      ["y", "Y", -2, 2, .01],
      ["z", "Z", -3, 3, .01],
      ["rx", "Rot X", -180, 180, 1],
      ["ry", "Rot Y", -180, 180, 1],
      ["rz", "Rot Z", -180, 180, 1],
      ["scale", "Scale", .35, 2.2, .01]
    ];

    function renderEditor() {
      const node = nodes.get(state.selected);
      selectedName.textContent = node.name;
      selectedKind.textContent = node.group;
      selectedPurpose.textContent = node.purpose;
      selectedSwatch.style.background = node.color;
      colorPicker.value = node.color;
      toggleVisible.textContent = node.visible ? "Hide" : "Show";
      controlsRoot.innerHTML = "";
      controlDefs.forEach(([key, label, min, max, step]) => {
        const row = document.createElement("div");
        row.className = "control";
        const value = document.createElement("span");
        value.className = "value";
        const input = document.createElement("input");
        input.type = "range";
        input.min = min;
        input.max = max;
        input.step = step;
        input.value = node.transform[key] ?? (key === "scale" ? 1 : 0);
        value.textContent = Number(input.value).toFixed(key === "scale" ? 2 : 1);
        input.addEventListener("input", () => {
          node.transform[key] = Number(input.value);
          value.textContent = Number(input.value).toFixed(key === "scale" ? 2 : 1);
        });
        const lab = document.createElement("label");
        lab.textContent = label;
        row.append(lab, input, value);
        controlsRoot.appendChild(row);
      });
    }

    document.querySelectorAll(".group-tabs button").forEach(button => {
      button.addEventListener("click", () => {
        document.querySelectorAll(".group-tabs button").forEach(b => b.classList.remove("active"));
        button.classList.add("active");
        state.filter = button.dataset.filter;
        renderList();
      });
    });

    document.getElementById("search").addEventListener("input", event => {
      state.search = event.target.value;
      renderList();
    });

    colorPicker.addEventListener("input", event => {
      const node = nodes.get(state.selected);
      node.color = event.target.value;
      selectedSwatch.style.background = node.color;
      renderList();
    });

    toggleVisible.addEventListener("click", () => {
      const node = nodes.get(state.selected);
      node.visible = !node.visible;
      toggleVisible.textContent = node.visible ? "Hide" : "Show";
    });

    document.getElementById("resetElement").addEventListener("click", () => {
      const node = nodes.get(state.selected);
      node.transform = { ...node.base, scale: 1 };
      node.visible = true;
      node.color = elementDefs.find(item => item.id === node.id).color;
      renderEditor();
      renderList();
    });

    document.getElementById("resetView").addEventListener("click", () => {
      state.yaw = -38 * DEG;
      state.pitch = 20 * DEG;
      state.distance = 13.5;
    });

    document.getElementById("explode").addEventListener("click", event => {
      state.exploded = !state.exploded;
      event.currentTarget.textContent = state.exploded ? "Assemble" : "Explode";
    });

    document.getElementById("spin").addEventListener("click", event => {
      state.propellers = !state.propellers;
      event.currentTarget.textContent = state.propellers ? "Stop Props" : "Propellers";
    });

    xray.addEventListener("change", () => {
      const fc = nodes.get("flight-controller");
      fc.meshes.forEach(m => {
        if (m.color === "#aee8ff") m.alpha = xray.checked ? .18 : .04;
      });
    });

    canvas.addEventListener("pointerdown", event => {
      canvas.setPointerCapture(event.pointerId);
      state.drag = { x: event.clientX, y: event.clientY, yaw: state.yaw, pitch: state.pitch };
    });

    canvas.addEventListener("pointermove", event => {
      if (!state.drag) return;
      const dx = event.clientX - state.drag.x;
      const dy = event.clientY - state.drag.y;
      state.yaw = state.drag.yaw + dx * .008;
      state.pitch = clamp(state.drag.pitch + dy * .006, -1.2, 1.25);
    });

    canvas.addEventListener("pointerup", event => {
      canvas.releasePointerCapture(event.pointerId);
      state.drag = null;
    });

    canvas.addEventListener("wheel", event => {
      event.preventDefault();
      state.distance = clamp(state.distance + event.deltaY * .012, 7, 32);
    }, { passive: false });

    renderList();
    renderEditor();
    requestAnimationFrame(renderLoop);
  </script>
